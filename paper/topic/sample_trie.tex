	{ #1 \& {} \& {} \& {} \& {} \\};
	\node[font=\tiny, anchor=south] at (n#1-1-2.north) {0};
	\node[font=\tiny, anchor=south] at (n#1-1-3.north) {1};
	\node[font=\tiny, anchor=south] at (n#1-1-4.north) {2};
	\node[font=\tiny, anchor=south] at (n#1-1-5.north) {3}
}
